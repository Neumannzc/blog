handleCommand(command, row) {
  switch (command) {
	case 'handleResetPwd':
	  this.handleMethodOne(row)
	  break
	case 'handleAuthRole':
	  this.handleMethodTwo(row)
	  break
	case 'handleBalance':
	  this.handleMethodThree(row)
	  break
	default:
	  break
  }
},
