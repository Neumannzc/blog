import org.springframework.stereotype.Component;

@Component
public class JwtUtil {

    @Value("${jwt.secret}")
    private String secret; // å¯†é’¥ï¼ˆéœ€ä»é…ç½®æ–‡ä»¶è¯»å–ï¼‰

    @Value("${jwt.expiration}")

    private String doGenerateToken(Map<String, Object> claims, long expiration) {

                .setClaims(claims)
                .setIssuedAt(new Date(System.currentTimeMillis()))
                .setExpiration(new Date(System.currentTimeMillis() + expiration))
                .signWith(SignatureAlgorithm.HS256, secret)
                .compact();
    }

    // è§£æ Token è·å– Claims
    public Claims parseToken(String token) {
        return Jwts.parser()
                .setSigningKey(secret)
                .parseClaimsJws(token)
                .getBody();
    }

    // éªŒè¯ Token æ˜¯å¦æœ‰æ•ˆ
    public boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(secret).parseClaimsJws(token);

        } catch (JwtException e) {
            return false;
        }
    }

    // è·å– Token ä¸­çš„æŒ‡å®šå£°æ˜
    public <T> T getClaim(String token, String claimName, Class<T> clazz) {
